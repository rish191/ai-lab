# Define initial states
monkey_pos = 'door'      # Monkey starts at the door
box_pos = 'window'       # Box (chair) is at the window
banana_pos = 'center'    # Bananas hang from the center
monkey_on_box = False
has_banana = False

# Step 1: Monkey walks to box
monkey_pos = box_pos     # Monkey moves to the window
print('Monkey walks to the window.')

# Step 2: Monkey pushes box to center
box_pos = banana_pos     # Move box to beneath bananas
monkey_pos = banana_pos  # Monkey moves with the box
print('Monkey pushes the box to the center.')

# Step 3: Monkey climbs the box
monkey_on_box = True
print('Monkey climbs onto the box.')

# Step 4: Monkey grasps the bananas
if monkey_on_box and box_pos == banana_pos:
    has_banana = True
    print('Monkey grasps the bananas!')
else:
    print('Monkey cannot reach the bananas yet.')

# Final Output
if has_banana:
    print('Success: The monkey can reach the banana.')
else:
    print('Failure: The monkey cannot reach the banana.')
